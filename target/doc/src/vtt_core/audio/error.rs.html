1//! Error types for audio processing
2
3use thiserror::Error;
4
5/// Result type for audio operations
6pub type AudioResult<T> = Result<T, AudioError>;
7
8/// Errors that can occur during audio processing
9#[derive(Error, Debug)]
10pub enum AudioError {
11    /// Audio device error (enumeration, initialization, etc.)
12    #[error("Audio device error: {0}")]
13    DeviceError(String),
14
15    /// Audio stream error (configuration, start/stop, etc.)
16    #[error("Audio stream error: {0}")]
17    StreamError(String),
18
19    /// Audio capture error
20    #[error("Audio capture error: {0}")]
21    CaptureError(String),
22
23    /// I/O error (file operations)
24    #[error("I/O error: {0}")]
25    IoError(#[from] std::io::Error),
26
27    /// Generic error
28    #[error("Audio error: {0}")]
29    Other(String),
30}
31
32impl From<cpal::DevicesError> for AudioError {
33    fn from(err: cpal::DevicesError) -> Self {
34        AudioError::DeviceError(err.to_string())
35    }
36}
37
38impl From<cpal::BuildStreamError> for AudioError {
39    fn from(err: cpal::BuildStreamError) -> Self {
40        AudioError::StreamError(err.to_string())
41    }
42}
43
44impl From<cpal::PlayStreamError> for AudioError {
45    fn from(err: cpal::PlayStreamError) -> Self {
46        AudioError::StreamError(err.to_string())
47    }
48}
